// মেসার্স শরিফ ডিস্ট্রিবিউশন হাউস — Apps Script v2 // পুরো কোড মুছে এটা পেস্ট করুন function doGet(e) { const action = e.parameter.action; let result; try { if (action === 'getSales') result = getSales(e.parameter); else if (action === 'addSale') result = addSale(e.parameter); else if (action === 'delSale') result = delSale(e.parameter); else if (action === 'getEmps') result = getEmps(); else if (action === 'addEmp') result = addEmp(e.parameter); else if (action === 'delEmp') result = delEmp(e.parameter); else if (action === 'getStock') result = getStock(); else if (action === 'updateStock') result = updateStock(e.parameter); else result = { error: 'Unknown action' }; } catch (err) { result = { error: err.toString() }; } const json = JSON.stringify(result); const callback = e.parameter.callback; if (callback) { return ContentService .createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { return doGet(e); } // ── SALES ────────────────────────────────────── function getSales(p) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('sales'); const data = sh.getDataRange().getValues(); if (data.length <= 1) return { sales: [] }; let rows = data.slice(1).map(r => { // date normalize করো — যেকোনো format থেকে YYYY-MM-DD let dateStr = r[1]+''; try { const d = new Date(r[1]); if (!isNaN(d)) dateStr = d.getFullYear()+'-'+String(d.getMonth()+1).padStart(2,'0')+'-'+String(d.getDate()).padStart(2,'0'); } catch(e) {} return { id:r[0]+'', date:dateStr, prod:r[2]+'', emp:r[3]+'', qty:+r[4], buy:+r[5], sell:+r[6], total:+r[7], cost:+r[8], profit:+r[9], cust:r[10]+'', note:r[11]+'' }; }); if (p.date) rows = rows.filter(r => r.date === p.date); if (p.emp) rows = rows.filter(r => r.emp === p.emp); return { sales: rows }; } function addSale(p) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('sales'); const id = Date.now()+''; const qty=+p.qty, buy=+p.buy||0, sell=+p.sell; // date কে YYYY-MM-DD format এ store করো const d = new Date(p.date); const dateStr = d.getFullYear()+'-'+String(d.getMonth()+1).padStart(2,'0')+'-'+String(d.getDate()).padStart(2,'0'); sh.appendRow([id, dateStr, p.prod, p.emp, qty, buy, sell, qty*sell, qty*buy, qty*(sell-buy), p.cust||'', p.note||'']); deductStock(p.prod, qty); return { success: true, id }; } function delSale(p) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('sales'); const data = sh.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0]+'' === p.id+'') { sh.deleteRow(i+1); return { success: true }; } } return { error: 'Not found' }; } // ── EMPLOYEES ────────────────────────────────── function getEmps() { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('employees'); const data = sh.getDataRange().getValues(); if (data.length <= 1) return { emps: [] }; return { emps: data.slice(1).map(r => ({ id:r[0]+'', name:r[1]+'', role:r[2]+'', phone:r[3]+'', area:r[4]+'', pin:r[5]+'' }))}; } function addEmp(p) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('employees'); const id = Date.now()+''; sh.appendRow([id, p.name, p.role, p.phone||'', p.area||'', p.pin||'0000']); return { success: true, id }; } function delEmp(p) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('employees'); const data = sh.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0]+'' === p.id+'') { sh.deleteRow(i+1); return { success: true }; } } return { error: 'Not found' }; } // ── STOCK ─────────────────────────────────────── function getStock() { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('stock'); const data = sh.getDataRange().getValues(); if (data.length <= 1) return { stock: [] }; return { stock: data.slice(1).map(r => ({ prod:r[0]+'', price:+r[1], qty:+r[2], restock:+r[3]||10 }))}; } function updateStock(p) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('stock'); const data = sh.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0]+'' === p.prod+'') { sh.getRange(i+1, 3).setValue(+p.qty); if (p.restock) sh.getRange(i+1, 4).setValue(+p.restock); return { success: true }; } } return { error: 'Product not found' }; } function deductStock(prod, qty) { try { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('stock'); const data = sh.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0]+'' === prod+'') { const cur = +data[i][2]; sh.getRange(i+1, 3).setValue(Math.max(0, cur - qty)); return; } } } catch(e) {} } // ── SETUP (একবার চালান) ───────────────────────── function setupSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const hStyle = sh => { sh.getRange(1,1,1,sh.getLastColumn()).setFontWeight('bold') .setBackground('#1e3a8a').setFontColor('#ffffff'); }; // sales let sh = ss.getSheetByName('sales') || ss.insertSheet('sales'); if (sh.getLastRow() === 0) { sh.appendRow(['id','date','prod','emp','qty','buy','sell','total','cost','profit','cust','note']); hStyle(sh); } // employees sh = ss.getSheetByName('employees') || ss.insertSheet('employees'); if (sh.getLastRow() === 0) { sh.appendRow(['id','name','role','phone','area','pin']); hStyle(sh); } // stock — নতুন sh = ss.getSheetByName('stock') || ss.insertSheet('stock'); if (sh.getLastRow() === 0) { sh.appendRow(['prod','price','qty','restock_alert']); hStyle(sh); const prods = [ ['Toylet White',18.50,100,20], ['Toylet Gold Red',25.50,100,20], ['Toylet Red Super',23.50,100,20], ['150 Hand Towel Cleaner',28.50,100,20], ['150 Hand Towel Premium',32.50,100,20], ['150 Hand Towel High',33.50,100,20], ['Toylet Napkin',44.50,100,20], ['Touch & Tech',42.00,100,20], ['250 Super',57.00,100,20] ]; prods.forEach(p => sh.appendRow(p)); } SpreadsheetApp.getUi().alert('✅ সেটআপ সম্পন্ন! এখন নতুন Deployment করুন।'); }
